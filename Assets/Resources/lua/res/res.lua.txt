res = res or {}
local _M = res

function _M.__init__()
    _M:InitView()
end

function _M:InitView()
    self.gameObject = UnityEngine.GameObject.Find("Canvas/Root_Panel/res").gameObject
    self.transform = self.gameObject.transform
    local transform = self.transform

    --
    self.grid = {
        gameObject = transform:Find("-grid").gameObject,
        Image_grid_player = self.transform:Find("-grid/$grid_player"):GetComponent("Image"),
        Image_grid_enemy = self.transform:Find("-grid/$grid_enemy"):GetComponent("Image"),
        Image_grid_attack = self.transform:Find("-grid/$grid_attack"):GetComponent("Image"),
        Image_grid_defense = self.transform:Find("-grid/$grid_defense"):GetComponent("Image"),
        Image_grid_hp = self.transform:Find("-grid/$grid_hp"):GetComponent("Image")
    }
    self.item = {
        gameObject = transform:Find("-item").gameObject,
        Image_item_player = self.transform:Find("-item/$item_player"):GetComponent("Image"),
        Image_item_enemy = self.transform:Find("-item/$item_enemy"):GetComponent("Image"),
        Image_item_attack = self.transform:Find("-item/$item_attack"):GetComponent("Image"),
        Image_item_defense = self.transform:Find("-item/$item_defense"):GetComponent("Image"),
        Image_item_hp = self.transform:Find("-item/$item_hp"):GetComponent("Image")
    }

    --
end

_M.__init__()
